using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Windows.Forms;

namespace WindowsFormsApplication1
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        private void button1_Click(object sender, EventArgs e)
        {
            double zahl1 = Convert.ToDouble(textBox1.Text);             //textbox gibt string wert zurück.muss deshalb in double umgewandelt werden
            double zahl2 = Convert.ToDouble(textBox2.Text);

            if (radioButton1.Checked == true)                              //schaut ob button wirklich ausgewählt
            {
                double erg = zahl1 + zahl2;                                // erg ist weitere variable (Ergebnis )  double wieder in string umwandeln
                textBox3.Text = Convert.ToString(erg);

            }

            else if (radioButton2.Checked == true)                              //schaut ob button wirklich ausgewählt
            {
                double erg = zahl1 - zahl2;
                textBox3.Text = Convert.ToString(erg);

            }

            if (radioButton3.Checked == true)                              //schaut ob button wirklich ausgewählt
            {
                double erg = zahl1 * zahl2;
                textBox3.Text = Convert.ToString(erg);

            }

            if (radioButton4.Checked == true)                              //schaut ob button wirklich ausgewählt
            {
                double erg = zahl1 / zahl2;
                textBox3.Text = Convert.ToString(erg);

            }
        }

        private void button3_Click(object sender, EventArgs e)             // Beenden Button
        {
            Close();
        }

        private void button2_Click(object sender, EventArgs e)               // Lösche Button. 
        {
            textBox1.Text = ("");
            textBox2.Text = ("");
            textBox3.Text = ("");
        }
    }
}
